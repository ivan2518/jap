using System;

namespace Ds
{
    internal class Sequence
    {
        public Sequence(int input_last)
        {
            for(int i = 0; i < input_last+1; i++) 
                Console.Write($"{i} ");

            Console.WriteLine();
        }
        public Sequence(int input_firs, int input_last) 
        {
            for (int i =input_firs; i != input_last+1; i++)
                Console.Write($"{i} ");

            Console.WriteLine();
        }
        public Sequence(char input_last)
        {
            int index_of_input_letter = letters.IndexOf(input_last);

            for (int i = 0; i != index_of_input_letter + 1; i ++)
                Console.Write($"{letters[i]} ");

            Console.WriteLine();
        }
        public Sequence(char input_first, char input_last)
        {
            int index_first_letter = letters.IndexOf(input_first);
            int index_last_letter = letters.IndexOf(input_last);

            for (int i = index_first_letter; i != index_last_letter+1; i ++)
                Console.Write($"{letters[i]} ");

            Console.WriteLine();
        }
        private string letters = "abcdefghijklmnopqrstuvwxyz";
    }

    internal class Program
    {
        static void Main(string[] args)
        {
            Sequence first_example = new Sequence(10);
            Sequence second_example = new Sequence(0, 10);
            Sequence thurd_example = new Sequence('z');
            Sequence fourth_example = new Sequence('a', 'z');
        }
    }
}
